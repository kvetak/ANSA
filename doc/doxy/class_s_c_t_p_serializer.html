{
    uint32 h;
    unsigned char byte0, byte1, byte2, byte3;
    uint32 crc32c;
    register uint32 i;
    register uint32 res = (~0L);
    for (i = 0; i < len; i++)
      CRC32C(res, buf[i]);
    h      = ~res;
    byte0  = h & 0xff;
    byte1  = (h>>8) & 0xff;
    byte2  = (h>>16) & 0xff;
    byte3  = (h>>24) & 0xff;
    crc32c = ((byte0 << 24) | (byte1 << 16) | (byte2 << 8) | byte3);
    return htonl(crc32c);
}
